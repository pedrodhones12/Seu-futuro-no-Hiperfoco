def apresentar_area(nome, descricao, talentos_tea, talentos_tdah, atividades):
    print("\n" + "-"*60)
    print(f"🌱 ÁREA: {nome.upper()}")
    print("-"*60)
    print(f"📝 O que é: {descricao}")
    print("\n✨ Seus talentos que usará aqui:")
    print(f"🔹 TEA: {talentos_tea}")
    print(f"🔹 TDAH: {talentos_tdah}")
    print("\n🎯 Atividades que você vai fazer:")
    for atv in atividades:
        print(f"   ➤ {atv}")
    time.sleep(2)
    print("\n💡 Você achou essa área interessante?")
    while True:
        resp = input("Digite S para SIM ou N para NÃO: ").strip().upper()
        if resp in ["S", "N"]:
            return resp
        print("Resposta inválida! Digite apenas S ou N.")

def etapa_aprofundamento(areas_interessantes):
    print("\n" + "="*70)
    print("🔍 ETAPA 3: APROFUNDAMENTO — ONDE O TALENTO CRESCE")
    print("="*70)
    time.sleep(1)
    print("Você gostou dessas áreas? Agora vamos explorar cada uma com muito mais detalhes!")
    print("Lembre-se: seu hiperfoco é a capacidade de ir muito mais fundo do que qualquer um imagina — é o seu dom!\n")
    preferidas = []
    for area in areas_interessantes:
        print(f"➡️ Vamos aprofundar em: {area['nome']}")
        print(f"   {area['descricao']}")
        resp = input("Você quer ficar mais tempo aqui? (S/N): ").strip().upper()
        if resp == "S":
            preferidas.append(area)
    return preferidas

def etapa_escolha_final(preferidas):
    print("\n" + "="*70)
    print("🤔 ETAPA 4: O CAMINHO DAS PREFERÊNCIAS")
    print("="*70)
    time.sleep(1)
    print("Agora vamos descobrir qual área é a que mais faz sua mente brilhar!")
    if len(preferidas) == 1:
        return preferidas[0]
    print("\nEscolha a área que você mais amou explorar:")
    for i, area in enumerate(preferidas, 1):
        print(f"{i}. {area['nome']}")
    while True:
        try:
            escolha = int(input("Digite o número da sua escolha: "))
            if 1 <= escolha <= len(preferidas):
                return preferidas[escolha - 1]
            else:
                print(f"Digite um número entre 1 e {len(preferidas)}")
        except ValueError:
            print("Digite apenas números!")
